[docs]def comp_specs(cubelist,reglist,figfile='comp_specs.pdf'):
    """
    + extract spectra from specified apertures from mutiple spectral cubes
    + plot spectra of mutiple spectral cube from the same regions in one panel
    """
    
    cube=SpectralCube.read(cubelist[0],mode='readonly')
    header=fits.getheader(cubelist[0])
    
    slab = cube.spectral_slab(45.50 * u.GHz, 45.70 * u.GHz)
    mom0=(slab.moment(order=0)).array
    
    plt.clf()
    figsize=(11,6)
    fig=plt.figure(figsize=figsize)
    
    w=WCS(header).celestial
    
    ax = fig.add_axes([-0.075,0.1,0.7,0.7],projection=w)
    ax.imshow(mom0, interpolation='nearest',
           vmin=np.nanmin(mom0), vmax=np.nanmax(mom0))
    xlim=ax.get_xlim()
    ylim=ax.get_ylim()
    
    xlim=(xlim[0]+0,xlim[1]-0)
    ylim=(ylim[0]+0,ylim[1]-0)
    ax.set_xlim(xlim)
    ax.set_ylim(ylim)
    ax.coords['ra'].set_axislabel('Right Ascension (ICRS)')
    ax.coords['dec'].set_axislabel('Declination (ICRS)')
    
    for i,reg in enumerate(reglist):
        
        reg_pix=reg.to_pixel(w)
        reg_pix.plot(ax=ax,facecolor='none', edgecolor='yellow', lw=0.5)
        reg_pix_text=regions.TextPixelRegion(reg_pix.center,str(i+1))
        reg_pix_text.plot(ax=ax,color='yellow')
    
    sclist=[]
    for cubename in cubelist:
        sclist.append(SpectralCube.read(cubename,mode='readonly'))
        
    for i,reg in enumerate(reglist):
        
        ax_res=fig.add_subplot(1,1,i+1)
        
        for j,sc in enumerate(sclist):
            
            subsc=sc.subcube_from_regions([reg])
            subsc_1d=subsc.sum(axis=(1,2))
            if j!=0:
                subsc_1d=subsc_1d*1.1
            
            freq=sc.spectral_axis.value/1e9
            area=1.0
            if  j==0:
                ax_res.fill_between(freq,0,subsc_1d.array/area,facecolor='0.8',step='mid',alpha=0.75)        
            #ax_res.plot(freq,subsc_1d.array/area,color='0.8',label='Data')            
            ax_res.plot(freq,subsc_1d.array/area,label=str(j),alpha=1.,lw=4.0/(2**j))
            ax_res.legend() 

    fig.subplots_adjust(left=0.50,bottom=0.10,right=0.95,top=0.80,wspace=0.05,hspace=0.05)
    for ax_sub in fig.get_axes():
        if  not isinstance(ax_sub,astropy.visualization.wcsaxes.core.WCSAxes):
            ax_sub.set_xlabel('Frequency [GHz]')
            ax_sub.set_ylabel('Flux [Jy]')
            ax_sub.label_outer()
            
    fig.savefig(figfile)   
    plt.close()
